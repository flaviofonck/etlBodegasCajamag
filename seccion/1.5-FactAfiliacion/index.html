    Validador de campos repetidos¶

    Se valida la presencia de registros duplicados en todas las tablas. Si no existe la columna cod, se crea concatenando las columnas numdoc y coddoc. A continuación, se comparan las columnas relevantes para detectar duplicados y se almacenan los resultados usando StoreDuplicated. Luego, se eliminan los duplicados usando RemoveDuplicated, y se registra el proceso en el log. Cualquier error se maneja de manera adecuada, registrando también los casos en que no se puede crear la columna cod.

    # Validador de campos repetidos
validador_time = time.time()

# Iterar sobre todas las tablas
for ky in [x for x in df_structure.keys()]:
    df = df_structure[ky]

    # Validar si la columna 'cod' existe; si no, crearla
    if 'cod' not in df.columns:
        if 'numdoc' in df.columns and 'coddoc' in df.columns:
            # Crear la columna 'cod' concatenando 'numdoc' y 'coddoc'
            df = df.assign(cod=df['numdoc'].astype(str) + df['coddoc'].astype(str))
            logger.info(f"Columna 'id' creada en el DataFrame: {ky}")
        else:
            # Si no existen las columnas necesarias para crear 'cod', saltar la tabla
            logger.error(f"No se encontraron las columnas 'numdoc' y 'coddoc' necesarias para crear 'id' en el DataFrame: {ky}")
            continue  # Saltar a la siguiente tabla si no se pueden crear los valores de 'cod'

    # Asegurarse de que la columna 'cod' existe antes de proceder
    if 'cod' in df.columns:
        # Definir las columnas para comparar, excluyendo 'id'
        ColumnsToCompare = [x for x in df.columns if x != 'cod']

        # Imprimir el nombre de la tabla en proceso
        print(f"Procesando tabla: {ky}")

        # Almacenar duplicados
        try:
            StoreDuplicated('cod', ColumnsToCompare, df, f'trazaDuplicados_{ky}')
        except KeyError as e:
            logger.error(f"Error en StoreDuplicated para la tabla {ky}: {str(e)}")
            continue

        # Limpieza inicial para quitar duplicados
        try:
            df_structure[ky] = RemoveDuplicated('cod', 'fecest', df)
        except KeyError as e:
            logger.error(f"Error en RemoveDuplicated para la tabla {ky}: {str(e)}")
            continue

        # Registrar la validación en el log
        logger.info(f'VALIDADOR TABLA: {ky}')
    else:
        logger.error(f"La columna 'cod' no fue creada correctamente en el DataFrame: {ky}")

    Definición de columnas con sus tipos¶

    Se clasifican las columnas de cada tabla en numéricas, fechas y texto. Las columnas que contienen "fec" en su nombre se convierten al tipo fecha, y se crean los periodos de afiliación y retiro en formato AñoMes. El proceso se registra en el log, incluyendo el tiempo total de ejecución.

    #Definir columnas con sus tipos
transfor2_time = time.time()
numerics = ['int16', 'int32', 'int64', 'float16', 'float32', 'float64']
numericColumns = dict()
datesColumns = dict()
textColumns = dict()

for ky in list( [x for x in df_structure.keys() ] ):
    numericColumns[ky] = df_structure[ky].select_dtypes( include = numerics ).columns.tolist()
    datesColumns[ky] = [x for x in df_structure[ky].columns.tolist() if x.startswith('fec') ]
    textColumns[ky] = [x for x in df_structure[ky].columns.tolist() if x not in ( numericColumns[ky] + datesColumns[ky] ) ]+ [x for x in df_structure[ky].columns.tolist() if x.startswith('cod') ]+ [x for x in df_structure[ky].columns.tolist() if x.startswith('ced') ]+ [x for x in df_structure[ky].columns.tolist() if x.startswith('num') ]
    textColumns[ky] = list(set(textColumns[ky]))
    #Convertir columnas que tengan "fec" en el nombre en tipo fecha
    df_structure[ky][datesColumns[ky]] = df_structure[ky][datesColumns[ky]].apply( lambda x: pd.to_datetime( x , errors='coerce' ) )

    #Se crean Los periodo para afiliación y retiro (corrección cuando hay NAN en afi o en ret)
    df_structure[ky]['perafi'] = (df_structure[ky]['perafi']).fillna(188001).astype(int).astype(str)
    df_structure[ky]['perret'] = (df_structure[ky]['perret']).fillna(300001).astype(int).astype(str)
